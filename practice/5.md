## Задание
Проверить, что код дуальный коду Хэмминга действительно является симплексом.

---

## Решение
```python
import itertools

def build_simplex_generator_columns(r: int) -> list[list[int]]:
  n = (1 << r) - 1
  return [[int(bit) for bit in format(value, f"0{r}b")] for value in range(1, n + 1)]

def encode_message(message: tuple[int, ...], columns: list[list[int]]) -> list[int]:
  return [sum(m * c for m, c in zip(message, column)) % 2 for column in columns]

def verify_dual_hamming_is_simplex(r: int) -> bool:
  n = (1 << r) - 1
  expected_weight = 1 << (r - 1)
  columns = build_simplex_generator_columns(r)
  messages = list(itertools.product((0, 1), repeat=r))

  print(f"=== Проверка для r = {r} ===")
  print(f"Параметры: n = {n}, ожидаемый вес ненулевых слов = {expected_weight}")
  print()

  all_ok = True

  for idx, message in enumerate(messages, start=1):
    codeword = encode_message(message, columns)
    weight = sum(codeword)
    label = "нулевое слово" if weight == 0 else "ненулевое слово"

    print(
      f"{idx:>2}. m = {message} -> c = {''.join(map(str, codeword))} | "
      f"w(c) = {weight} | {label}"
    )

    if weight not in (0, expected_weight):
      all_ok = False

  print()
  if all_ok:
    print(
      f"Итог: для r = {r} все ненулевые кодовые слова имеют вес {expected_weight}. "
      f"Значит, дуальный код Хэмминга является симплексным."
    )
  else:
    print(
      f"Итог: для r = {r} найдены слова с весом, отличным от {expected_weight}. "
      f"Следовательно, код не является симплексным."
    )
  print("-" * 72)
  print()
  return all_ok


for r in (3, 4, 5):
  verify_dual_hamming_is_simplex(r)
```
